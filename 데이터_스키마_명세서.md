## 🎯 이벤트 처리 명세

### 게임 미션 완료 이벤트

```javascript
// 미션 진행 이벤트
document.addEventListener('click', (e) => {
  if (e.target.classList.contains('btn-progress')) {
    const gameIndex = parseInt(e.target.getAttribute('data-index'));
    let state = getGlobalState();
    if (state.gameStates[gameIndex] < 3) {
      state.gameStates[gameIndex]++;
      saveGlobalState(state);
    }
  }
});

// 리워드 수령 이벤트
document.addEventListener('click', (e) => {
  if (e.target.classList.contains('btn-receive')) {
    const gameIndex = parseInt(e.target.getAttribute('data-index'));
    let state = getGlobalState();
    state.gameStates[gameIndex] = 4;
    saveGlobalState(state);
    addReward('usdt', 2, '게임 미션 완료 보상');
    showUsdtReward(2, '게임 미션 보상을 받았습니다');
  }
});
```

### 출석 체크 이벤트

```javascript
document.getElementById('btn-attend').addEventListener('click', () => {
  if (!canAttend || currentDay > 5) return;
  
  const reward = rewards[currentDay];
  if (reward.type === 'usdt') {
    addReward('usdt', reward.amount, '출석 체크 보상');
    showUsdtReward(reward.amount, '내일도 출석 체크하고 보상 받으세요');
  } else if (reward.type === 'ticket') {
    addReward('ticket', reward.amount, '');
    showTicketReward(reward.amount);
  }
  
  canAttend = false;
  renderAttendance();
});
```
